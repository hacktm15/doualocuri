use doi_pasi;

drop table pubs;
--drop table utilizatori;
drop table mese;
drop table reservation;

CREATE TABLE PUBS(

id int(10) PRIMARY KEY AUTO_INCREMENT,
name character(30) not null ,
descriere blob,
lat float not null,
lng float not null,
adresa character(100));

create table mese(

id int(10) primary key AUTO_INCREMENT,
id_ul_local_al_mesei int(10),
pub_id int(10) references pubs(id),
locuri int(10),
is_free boolean,
zona character(10));

create table reservation(
id int(10) primary key AUTO_INCREMENT,
table_id int(10) references mese(id),
moment timestamp
);
